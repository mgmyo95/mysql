DELIMITER // 

    CREATE FUNCTION totalamount_fun()
    RETURNS DECIMAL 
        BEGIN 
            DECLARE result DECIMAL;
            SELECT SUM(products.price * checkouts.qty)
            INTO result 
            FROM checkouts 
            LEFT JOIN products 
            ON checkouts.product_id = products.id;
            RETURN result;
        END;//
